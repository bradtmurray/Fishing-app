            const searchTerm = document.getElementById('fishSearch').value.toLowerCase();

            const filteredFish = fishDatabase.filter(fish => 

                fish.name.toLowerCase().includes(searchTerm) ||

                fish.spots.some(spot => spot.toLowerCase().includes(searchTerm))

            );

            

            displayFishCards(filteredFish);
